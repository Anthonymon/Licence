\section{Graphe d'interférence}

\begin{frame}[fragile]
  \frametitle{Exemple de fonction}
  \begin{code}
    int f(int c) {
      int a,b;

      a = 0;
      while(a < N) {
        b=a+1;
        c=c+b;
        a=2*b;
      }
      return c;
    }
  \end{code}
\begin{itemize}
\item On pourrait mettre à jour les valeurs des variables sur la pile.
\item Les registres sont plus rapides !
\item Utiliser les registres comme variables temporaires.
\end{itemize}
\end{frame}

    \item Graphe d'interférence: deux noeuds sont connectés s'ils sont vivants

  \end{itemize}

\begin{frame}
  \frametitle{Equations de flots}
$$in(N) = use(N) \cup (out(N) - def(N))$$
$$out(N) = \bigcup_{s \in succ(N)} in(S)$$

\begin{itemize}
  \item Les variables utilis\'ees dans un noeud sont vivantes en entr\'ee
  \item Les variables vivantes en sortie d'un noeud sont soit vivantes en entr\'ee, soit d\'efinies dans le noeud. 
  \item Les variables vivantes en entr\'ee d'un noeud sont n\'ecessairement vivantes en sortie des
  noeud pr\'ec\'edents.
\end{itemize}
\end{frame}

  \begin{itemize}

    \item Règle de coloriage: les noeuds connectés doivent être affectés des

    \end{itemize}

 \begin{minipage}{.45\textwidth}
\begin{code}
   %edx = 0
L1:%edx = %edx + 1
   %eax = %eax + %edx
   %edx = %edx * 2
   if %edx < N goto L1
   return %eax
\end{code}
\end{minipage}
\end{frame}

\begin{frame}[fragile]
  \frametitle{Coloriage de graphe}
  \begin{itemize}
    \item Si on dispose de k registres physiques, le problème d'allocation de
      registres
      est équivalent au problème de coloriage du graphe d'interférence avec k
      couleurs.
    \item NP-complet.
    \item Mais heuristique performante: \alert{coloriage par simplification.}
    \end{itemize}
\end{frame}

\begin{frame}
\frametitle{Simplification}
\begin{lemma}
  Soit un graphe $G$ et un de ses noeuds $N$.
  On veut colorier $G$ avec $K$ couleurs.
  Si le degré de $N$ est strictement inférieur à $K$, alors
  si on peut colorier $G\backslash N$, on peut
  colorier $G$.
\end{lemma}

\section{Spilling}
\begin{frame} \frametitle{Problème}
  \begin{itemize}
  \item L'heuristique ci-dessus peut ne pas fonctionner:
    \begin{itemize}
      \item Lors de la phase Simplify tous les noeuds sont de degré $\geq K$.
    \end{itemize}
  \item Solution: \structure{Spilling de registres}
  \end{itemize}

\end{frame}

\begin{frame} \frametitle{Spilling}
  \begin{itemize}
  \item Pas assez de registres.
  \item Il faut diminuer le degré d'une variable $a$ dans le graphe d'interférence.
  \item On alloue une case sur la pile.
  \item Chaque fois que l'on veut utiliser $a$, on le charge depuis la pile.
  \item Après utilisation on le sauve sur la pile.
  \end{itemize}
\end{frame}

\begin{frame} \frametitle{Simplification avec Spilling}
  \begin{tikzpicture}[node distance = 2cm]
    \node (B) {Build};
    \node[right of=B] (S) {Simplify};
    \node[right of=S, text width=1.5cm] (P) {Potential Spill ?};
    \node[right of=P] (C) {Select};
    \node[right of=C] (A) {Actual Spill ?};

    \draw[->] (B) -- (S);
    \draw[->] (S) -- (P);
    \draw[->] (P) -- (C);
    \draw[->] (C) -- (A);

    \draw[->] (S) edge[loop above] (S);
    \draw[->] (P) edge[bend left = 60] (S);
    \draw[->] (C) edge[loop above] (C);
    \draw[->] (A) edge[bend left = 60] node[auto] {Rebuild the graph} (B);
   \end{tikzpicture}

   \begin{itemize}
     \item Si un noeud de degré supérieur à K apparait lors de la
       simplification. On enregistre un spill possible.
     \item Parfois lors de la coloration on peut tout de même colorier
       les spill possibles. Si ce n'est pas le cas, on a un vrai spill.
     \item Dans ce cas on ``spill'' le registre sur la pile et on recommence.
   \end{itemize}
\end{frame}


\begin{frame} \frametitle{Quel noeud faut il spiller ?}
   \begin{itemize}
     \item Un noeud qui éliminera beaucoup d'arcs d'interférence.
     \item Un noeud utilisé peu fréquemment (le coût de passage par la pile est
       non négligeable). Éviter de spiller une variable d'induction dans une
       boucle !
   \end{itemize}
\end{frame}

\begin{frame} \frametitle{Fusion de registres}
\begin{itemize}
\item Dans le graphe d'interférence on peut fusionner deux sommets:
  \begin{itemize}
    \item ce sont des copies
    \item ils n'interfèrent pas entre eux.
  \end{itemize}
\item Dans ce cas:
  \begin{itemize}
    \item Les arcs du nouveau sommet sont l'union des arcs des anciens sommets.
    \item On supprimme l'instruction de copie du programme initial.
  \end{itemize}
\end{itemize}
\end{frame}

\begin{frame} \frametitle{Quand fusionner}
\begin{itemize}
\item La fusion peut rendre le graphe non K-coloriable.
\item Deux stratégies de fusion entre $a$ et $b$:
  \begin{itemize}
    \item Briggs: $ab$ a moins de K voisins de degrée $\geq K$.
    \item George: tous les voisins de $a$ sont
      \begin{itemize}
        \item de degré $< K$
        \item ou déjà en interférence avec $b$
      \end{itemize}
  \end{itemize}
\end{itemize}
\end{frame}

\section{Autres subtilités}
\begin{frame}\frametitle{Noeuds précoloriés}
  \begin{itemize}
    \item Certains registres physiques sont utilisés pour des buts précis:
      retour d'une fonction, pointeur de pile, etc.
    \item Pour épargner ces registres on peut précolorier certains noeuds
      du graphe d'inférence.
    \item Un sommet précolorié ne sera jamais simplifié.
  \end{itemize}
\end{frame}

\begin{frame}\frametitle{Registres et appels de fonctions}
  \begin{itemize}
    \item Certains registres sont:
      \begin{itemize}
        \item callee-save: à sauver par l'appelé
        \item caller-save: à sauver par l'appelant
      \end{itemize}
    \item Une instruction call redéfinit tous les caller-save
    \item On privilégiera:
      \begin{itemize}
        \item les caller-save pour les variables temporaires éphémères.
          \begin{itemize}
            \item pas besoin de faire un backup avant l'appel de fonction.
          \end{itemize}
        \item les callee-save pour les variables temporaires vivantes autour
          d'appels de sous programme.
      \end{itemize}
  \end{itemize}
\end{frame}

\section{Conclusion}
\begin{frame}\frametitle{Conclusion}
  \begin{itemize}
    \item Allocation de registre = Coloriage de graphe
    \item Heuristique efficace : build / simplify / select
    \item De nombreuses optimisations peuvent être utilisés pour:
      éliminer les copies, préserver certains registres, éviter les spills, etc.
  \end{itemize}
\end{frame}
